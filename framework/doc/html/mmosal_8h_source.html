1024static inline bool mmosal_safer_strcpy(char *dst, const char *src, size_t size)
1025{
1026 bool ret;
1027
1028 if (size == 0)
1029 {
1030 return true;
1031 }
1032
1033 strncpy(dst, src, size);
1034
1035 ret = dst[size-1] != '\0';
1036 dst[size-1] = '\0';
