118static inline char mm_nibble_to_hex_char(uint8_t nibble)
119{
120 nibble &= 0x0f;
121 if (nibble < 0x0a)
122 {
123 return '0' + nibble;
124 }
125 else
126 {
127 return 'A' + nibble - 0x0a;
128 }
129}
